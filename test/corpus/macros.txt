===============
nrm macro -- scratch register
===============
nrm r0, r1
---------------
dp4 r0, r1, r1
rsq r0, r0
mul r0, r1, r0
===============
nrm macro -- uniform register
===============
nrm r0, c1
---------------
mov r0, c1
dp4 r0, c1, r0
rsq r0, r0
mul r0, c1, r0
===============
frc instruction -- uniform register
===============
frc r0, c1
---------------
flr r0, c1
add r0, c1, -r0
===============
sgn instruction -- scratch register
===============
sgn r0, r1, r2, r3
sgn r0, -r1, r2, r3
---------------
slt r2, -r1, r1
slt r3, r1, -r1
add r0, r2, -r3
slt r2, r1, -r1
slt r3, -r1, r1
add r0, r2, -r3
===============
sgn instruction -- uniform register
===============
sgn r0, c1, r2, r3

sgn r0, -c1, r2, r3
---------------
mov r2, -c1
slt r2, -c1, -r2
mov r3, c1
slt r3, c1, -r3
add r0, r2, -r3

mov r2, c1
slt r2, c1, -r2
mov r3, -c1
slt r3, -c1, -r3
add r0, r2, -r3
===============
pow macro -- scratch register
===============
pow r0, r1, r2
---------------
max r0, r1, -r1
lg2 r0, r0
mul r0, r2, r0
ex2 r0, r0
